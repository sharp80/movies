			ts[i].prop = {name:ts[i].prop,renderer:this.propRenderer(ts[i].prop)};
			ts[i].target = typeof(ts[i].target)=="string"?document.getElementById(ts[i].target):ts[i].target;
			ts[i].easing = ts[i].easing?ts[i].easing:Tween.easing.easeInOut;
			
		}

		if (this.initObj.delay){
			this.interval = setInterval("Tween.proxyIntervalDelay("+this.id+")",this.initObj.delay*1000);
			if (this.initObj.count>0) this.initObj.count--;
		} else if (this.initObj.autoStart) {
			this.dispatchFinishedStart(1);
			this.startInterval();
		} 
	}

	Tween.prototype.startInterval = function(){
		this.motion = true;		
		this.timer = 0;
		this.interval = setInterval("Tween.proxyInterval("+this.id+")",Tween.intTime);
		
		this.percent = 0;
		this.iterator = 1/(this.initObj.time*Tween.fps);
		this.timeUnit = this.iterator*this.initObj.time;
		
	}

	Tween.prototype.processor = function(){
		if (this.percent>1) {
			this.stopMotion();
			this.dispatchFinishedStart(0);
			if (this.initObj.del){
				this.deleteTween();
				return;
			}
			this.reset();
		} else {
			this.applyEasing();
			this.dispatchIntervalFunctions();
		}
	}
	
	Tween.prototype.stop = function(){
		this.motion = false;
		clearInterval(this.interval);
		this.dispatchFinishedStart(0);
		this.reset();
	}	
	
	Tween.prototype.start = function(){
		if (!this.motion) {
			this.startInterval();
		}
	}	

	Tween.prototype.stopMotion = function(){
		this.motion = false;
		clearInterval(this.interval);
	}

	Tween.prototype.abort = function(kill){
		clearInterval(this.interval);
		this.motion = false;
		this.initObj.count=0;
		this.reset();
		if (kill) {this.deleteTween();}
	}

	Tween.prototype.deleteTween = function(){
		delete Tween.allTweens[this.id];
	}

	Tween.prototype.dispatchFinishedStart = function(startorfinish){
		var evtStr = startorfinish==0?"onFinish":"onStart";
		var f = this.initObj[evtStr];
		if (f){
			if (!f.scope) f.scope = this;
			if (!f.args) f.args=[];
			var trgArr = [];
			var i;
			for (i=0;i<this.initObj.targets.length;i++){
				trgArr.push(this.initObj.targets[i].target);
			}
			f.func.apply(f.scope,[trgArr].concat(f.args));	
		} 
			
		if (this.initObj.targets.length>0){
			var ts = this.initObj.targets;
			for (i=0;i<ts.length;i++){
				var f = ts[i][evtStr];
				if (f){
					f.func.apply(f.scope,[ts[i].target].concat(f.args));
				}
			}				
		}			
	}

	Tween.prototype.dispatchIntervalFunctions = function(){
		if (this.initObj.targets.length>0){
			var ts = this.initObj.targets;
			for (i=0;i<ts.length;i++){
				var f = ts[i]["onEnterFrame"];
				if (f){
					f.func.apply(f.scope,[ts[i].target].concat(f.args));
				}
			}				
		}		
	}
	
	Tween.prototype.reset = function() {
		this.percent = 0;
		this.timer = 0;
		if (this.initObj.count>1){
			this.initObj.count--;
			this.start();
			
		}
	}
	
	Tween.prototype.applyEasing = function(){
		var io = this.initObj;
		for (i=0;i<io.targets.length;i++){
			var val = io.targets[i].easing(this.timer,io.targets[i].start,io.targets[i].end-io.targets[i].start,io.time*Tween.fps);
			io.targets[i].prop.renderer(io.targets[i].target,io.targets[i].prop.name,val);
		}

		this.percent+=this.iterator;
		this.timer+=1;
		
	}
	
	Tween.prototype.propRenderer = function(prop){
		if (prop=="undefined") return function(){};
		switch(prop){
			case "opacity":
				return renderer_opacity;
				break;
			default:
				return renderer_position;
				break;
		}
		
		
		function renderer_opacity(target,prop,val){
		
			if (target.style) {
				if (target.style.filter){
					target.style.filter = 'alpha(opacity=' + val*100 + ')';
					target.style.opacity = val;
					//target.style.removeAttribute("filter");
				}else{
					target.style.opacity = val;
				}
			} else {
				this.stopMotion();
				this.reset();
			}
		}
		function renderer_position(target,prop,val){
			if (target.style) {
				target.style[prop] = Math.round(val)+"px";
			} else {
				this.stopMotion();
				this.reset();
			}
		}		
	}
